CREATE TABLE SON_PROJECT(
    PROJID NUMBER(4) NOT NULL CONSTRAINT PK_SON_PROJ PRIMARY KEY,
    P_DESC VARCHAR2(20) NOT NULL,
    P_START_DATE DATE NOT NULL ,
    P_END_DATE DATE NOT NULL ,
    BUDGET_AMOUNT NUMBER (7,2) NOT NULL ,
    MAX_NO_STAFF NUMBER (2)NOT NULL
    check ( P_START_DATE > P_END_DATE )
);
